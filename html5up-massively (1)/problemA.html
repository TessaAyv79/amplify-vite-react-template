import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt
from sklearn.feature_selection import SelectKBest, f_classif
from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import classification_report, confusion_matrix

# Load the CSV file into a DataFrame
df2 = pd.read_csv('C:/Users/Admin/Documents/MLAI/CSCN8030/Proje6/problem2.csv')

# Fill missing values for 'weight' with the mean
df2['weight'] = df2['weight'].fillna(df2['weight'].mean())

# Ensure 'days_before' is treated as a string before replacement
df2['days_before'] = df2['days_before'].astype(str).str.replace("days", "").astype(int)

# Standardize 'day_of_week' column
df2['day_of_week'] = df2['day_of_week'].replace({
    'Wednesday': 'Wed', 'Fri.': 'Fri', 'Monday': 'Mon'
})

# Replace '-' in 'category' column
df2['category'] = df2['category'].replace("-", "unknown")

# Convert categorical columns to dummy variables
df2 = pd.get_dummies(df2, columns=['day_of_week', 'time', 'category'])

# Check for any remaining missing values
print("Missing values after preprocessing:\n", df2.isnull().sum())

# Verify the types of columns
print("Data types of columns:\n", df2.dtypes)

# Ensure all columns are numeric (optional but useful for debugging)
print("Sample data:\n", df2.head())

# Feature Selection
# Prepare the features and target variable
X = df2.drop('attended', axis=1)
y = df2['attended']

# Verify that X contains only numerical values
print("Feature data types:\n", X.dtypes)

# Select the top 5 features
k = 5
selector = SelectKBest(f_classif, k=k)
X_new = selector.fit_transform(X, y)

# Print the selected features
selected_features = X.columns[selector.get_support()]
print("Selected features:", selected_features)

# Model Training and Evaluation
# Split the data into training and test sets
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

# Train a Random Forest Classifier
model = RandomForestClassifier()
model.fit(X_train, y_train)

# Make predictions and evaluate the model
y_pred = model.predict(X_test)
print(confusion_matrix(y_test, y_pred))
print(classification_report(y_test, y_pred))
